// Pattern Recognition Game
let patternCurrentLevel = 0;
let patternScore = 0;
const patterns = [
    { sequence: [2, 4, 6, 8, '?'], answer: 10, options: [9, 10, 12, 14, 16] },
    { sequence: [1, 4, 9, 16, '?'], answer: 25, options: [20, 24, 25, 30, 36] },
    { sequence: [5, 10, 20, 40, '?'], answer: 80, options: [60, 70, 80, 90, 100] },
    { sequence: [100, 95, 90, 85, '?'], answer: 80, options: [75, 80, 85, 90, 95] },
    { sequence: [3, 6, 12, 24, '?'], answer: 48, options: [36, 42, 48, 54, 60] }
];

function startPatternGame() {
    document.getElementById('pattern-game').style.display = 'block';
    patternCurrentLevel = 0;
    patternScore = 0;
    showPattern();
}

function showPattern() {
    if (patternCurrentLevel >= patterns.length) {
        document.getElementById('pattern-question').innerHTML = '
üéâ All Patterns Complete!
';
        document.getElementById('pattern-grid').innerHTML = '';
        return;
    }
    
    const pattern = patterns[patternCurrentLevel];
    document.getElementById('pattern-question').innerHTML = `
        Find the next number in the sequence:

        ${pattern.sequence.join(' ‚Üí ')}
    `;
    
    const grid = document.getElementById('pattern-grid');
    grid.innerHTML = '';
    
    pattern.options.forEach(option => {
        const cell = document.createElement('div');
        cell.className = 'pattern-cell';
        cell.textContent = option;
        cell.onclick = () => checkPattern(option, cell);
        grid.appendChild(cell);
    });
    
    document.getElementById('pattern-score').innerHTML = `Score: ${patternScore}/${patterns.length}`;
}

function checkPattern(selected, cell) {
    const pattern = patterns[patternCurrentLevel];
    const cells = document.querySelectorAll('.pattern-cell');
    cells.forEach(c => c.style.pointerEvents = 'none');
    
    if (selected === pattern.answer) {
        cell.classList.add('correct');
        patternScore++;
    } else {
        cell.classList.add('wrong');
        cells.forEach(c => {
            if (parseInt(c.textContent) === pattern.answer) {
                c.classList.add('correct');
            }
        });
    }
    
    setTimeout(() => {
        patternCurrentLevel++;
        showPattern();
    }, 1500);
}

function showChartScenario() {
    if (chartCurrentQuestion >= chartScenarios.length) {
        document.getElementById('chart-scenario').innerHTML = '
üéâ Perfect Matching!
';
        document.getElementById('chart-options').innerHTML = '';
        return;
    }
    
    const scenario = chartScenarios[chartCurrentQuestion];
    document.getElementById('chart-scenario').innerHTML = `
        Scenario ${chartCurrentQuestion + 1}:
${scenario.scenario}
    `;
    
    const optionsDiv = document.getElementById('chart-options');
    optionsDiv.innerHTML = '';
    
    scenario.options.forEach(option => {
        const optionDiv = document.createElement('div');
        optionDiv.className = 'chart-option';
        optionDiv.innerHTML = `
            
            
${option.name}

        `;
        optionDiv.onclick = () => checkChart(option.type, optionDiv);
        optionsDiv.appendChild(optionDiv);
    });
    
    document.getElementById('chart-score').innerHTML = `Score: ${chartScore}/${chartScenarios.length}`;
}

function checkChart(selected, element) {
    const scenario = chartScenarios[chartCurrentQuestion];
    const options = document.querySelectorAll('.chart-option');
    options.forEach(opt => opt.style.pointerEvents = 'none');
    
    if (selected === scenario.correct) {
        element.style.borderColor = '#00ff00';
        element.style.background = 'rgba(0, 255, 0, 0.2)';
        chartScore++;
    } else {
        element.style.borderColor = '#ff0000';
        element.style.background = 'rgba(255, 0, 0, 0.2)';
        options.forEach(opt => {
            if (opt.querySelector('div').textContent === scenario.options.find(o => o.type === scenario.correct).name) {
                opt.style.borderColor = '#00ff00';
                opt.style.background = 'rgba(0, 255, 0, 0.2)';
            }
        });
    }
    
    setTimeout(() => {
        chartCurrentQuestion++;
        showChartScenario();
    }, 2000);
}

const cleaningDatasets = [
    {
        data: [
            ['Name', 'Age', 'Email', 'Salary'],
            ['John', '25', 'john@email.com', '50000'],
            ['Sarah', 'NULL', 'sarah@email.com', '60000'],
            ['John', '25', 'john@email.com', '50000'],

        ],
        errors: [[2, 1], [3, 0], [4, 2], [5, 3]]
    }
];

let cleaningFound = [];
let cleaningDataset = 0;

function startCleaningGame() {
    document.getElementById('cleaning-game').style.display = 'block';
    cleaningDataset = 0;
    cleaningFound = [];
    showCleaningData();
}

function showCleaningData() {
    const dataset = cleaningDatasets[cleaningDataset];
    const grid = document.getElementById('cleaning-grid');

    let tableHTML = '';
    dataset.data.forEach((row, rowIndex) => {
        tableHTML += '';
        row.forEach((cell, colIndex) => {
            const isError = dataset.errors.some(([r, c]) => r === rowIndex && c === colIndex);
            const cellId = `cell-${rowIndex}-${colIndex}`;
            tableHTML += ``;
        });
        tableHTML += ''; // ‚úÖ moved outside the inner loop
    });
    tableHTML += '
${cell}
';

    grid.innerHTML = tableHTML;
    document.getElementById('cleaning-score').innerHTML =
        `Found: ${cleaningFound.length}/${dataset.errors.length} errors`;
}

function clickCell(row, col, isError) {
    const cellId = `cell-${row}-${col}`;
    const cell = document.getElementById(cellId);
    
    if (isError && !cleaningFound.some(([r, c]) => r === row && c === col)) {
        cell.classList.add('found');
        cleaningFound.push([row, col]);
        
        const dataset = cleaningDatasets[cleaningDataset];
        document.getElementById('cleaning-score').innerHTML = `Found: ${cleaningFound.length}/${dataset.errors.length} errors`;
        
        if (cleaningFound.length === dataset.errors.length) {
            setTimeout(() => {
                alert('üéâ All errors found! Great job cleaning the data!');
            }, 500);
        }
    }
}

let debugCurrent = 0;
let debugScore = 0;

function startDebugGame() {
    document.getElementById('debug-game').style.display = 'block';
    debugCurrent = 0;
    debugScore = 0;
    showDebugChallenge();
}

function showDebugChallenge() {
    if (debugCurrent >= debugChallenges.length) {
        document.getElementById('debug-question').innerHTML = '
üéâ All Bugs Found!
';
        document.getElementById('debug-code').style.display = 'none';
        document.getElementById('debug-answer').style.display = 'none';
        document.querySelector('#debug-game .game-btn').style.display = 'none';
        return;
    }
    
    const challenge = debugChallenges[debugCurrent];
    document.getElementById('debug-question').innerHTML = `Debug Challenge ${debugCurrent + 1}: What's wrong with this code?`;
    document.getElementById('debug-code').textContent = challenge.code;
    document.getElementById('debug-answer').value = '';
    document.getElementById('debug-result').innerHTML = '';
    document.getElementById('debug-score').innerHTML = `Score: ${debugScore}/${debugChallenges.length}`;
}

function checkDebug() {
    const userAnswer = document.getElementById('debug-answer').value.toLowerCase();
    const challenge = debugChallenges[debugCurrent];
    const isCorrect = challenge.answer.some(ans => userAnswer.includes(ans));
    
    const resultDiv = document.getElementById('debug-result');
    
    if (isCorrect) {
        debugScore++;
        resultDiv.innerHTML = '
‚úÖ Correct! You found the bug!
';
    } else {
        resultDiv.innerHTML = `
‚ùå Not quite. The issue is: ${challenge.issue}
`;
    }
    
    setTimeout(() => {
        debugCurrent++;
        showDebugChallenge();
    }, 3000);
}

        answer: 40,

let metricCurrent = 0;
let metricScore = 0;
let metricStartTime = null;
let metricTimes = [];

function startMetricRace() {
    document.getElementById('metric-race').style.display = 'block';
    metricCurrent = 0;
    metricScore = 0;
    metricTimes = [];
    metricStartTime = Date.now();
    showMetricQuestion();
}

function showMetricQuestion() {
    if (metricCurrent >= metricQuestions.length) {
        const avgTime = (metricTimes.reduce((a, b) => a + b, 0) / metricTimes.length).toFixed(1);
        const totalTime = ((Date.now() - metricStartTime) / 1000).toFixed(1);
        
        document.getElementById('metric-question').innerHTML = '
üèÅ Race Complete!
';
        document.getElementById('metric-answer').style.display = 'none';
        document.querySelector('#metric-race .game-btn').style.display = 'none';
        document.getElementById('metric-stats').innerHTML = `
            

                
${metricScore}/${metricQuestions.length}

                
Correct

            

            

                
${totalTime}s

                
Total Time

            

            

                
${avgTime}s

                
Avg Time

            

        `;
        return;
    }
    
    const questionStartTime = Date.now();
    const question = metricQuestions[metricCurrent];
    document.getElementById('metric-question').innerHTML = `Question ${metricCurrent + 1}:
${question.question}`;
    document.getElementById('metric-answer').value = '';
    document.getElementById('metric-result').innerHTML = '';
    
    window.metricQuestionStart = questionStartTime;
}

function checkMetric() {
    const userAnswer = parseFloat(document.getElementById('metric-answer').value);
    const question = metricQuestions[metricCurrent];
    const isCorrect = Math.abs(userAnswer - question.answer) <= question.tolerance;
    
    const questionTime = ((Date.now() - window.metricQuestionStart) / 1000).toFixed(1);
    metricTimes.push(parseFloat(questionTime));
    
    const resultDiv = document.getElementById('metric-result');
    
    if (isCorrect) {
        metricScore++;
        resultDiv.innerHTML = `
‚úÖ Correct! Time: ${questionTime}s
`;
    } else {
        resultDiv.innerHTML = `
‚ùå Wrong. Answer: ${question.answer}. Time: ${questionTime}s
`;
    }
    
    // Continue after a short delay
    setTimeout(() => {
        metricCurrent++;
        if (metricCurrent < metricQuestions.length) {
            loadMetricQuestion(); // show next question
